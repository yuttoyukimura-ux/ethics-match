<script>
  /* ここから裏側の仕組み（JavaScript）です */
  
  // 質問と、選んだ時にどの宗教に何点入るかの設定
  const questions = [
    {
      text: "この世界の成り立ちについて、どう思う？",
      choices: [
        { text: "絶対的な神が世界を創造したと思う", scores: { buddhism: 0, christianity: 1, islam: 1 } },
        { text: "神が創ったというより、あらゆるものは繋がり変化し続けている", scores: { buddhism: 2, christianity: 0, islam: 0 } }
      ]
    },
    {
      text: "人生で一番大切にすべきルールは？",
      choices: [
        { text: "神の愛を信じ、他者に無償の愛を注ぐこと", scores: { buddhism: 0, christianity: 2, islam: 0 } },
        { text: "神が定めた絶対的な戒律（ルール）を日常生活で厳格に守ること", scores: { buddhism: 0, christianity: 0, islam: 2 } },
        { text: "自分の欲望（煩悩）をコントロールし、心の平穏を保つこと", scores: { buddhism: 2, christianity: 0, islam: 0 } }
      ]
    }
  ];

  // 結果の解説テキスト
  const results = {
    buddhism: { name: "仏教", desc: "あなたの思想は「仏教」に近いかも！苦しみの原因は自分の心（煩悩）にあると考え、修行や自己コントロールによって悟りを目指すストイックな一面があります。四法印や八正道などの考え方がしっくりくるはず！" },
    christianity: { name: "キリスト教", desc: "あなたの思想は「キリスト教」に近いかも！神の愛（アガペー）を信じ、隣人を自分のように愛するという、温かく包み込むような精神を大切にしています。イエスの山上の垂訓などを読んでみると面白いかも！" },
    islam: { name: "イスラム教", desc: "あなたの思想は「イスラム教」に近いかも！唯一絶対の神（アッラー）に完全な服従を誓い、生活のあらゆる場面でルールをきっちり守る規律正しさを持っています。六信五行のシステムはあなたの性に合っているかも！" }
  };

  let currentQuestionIndex = 0;
  let totalScores = { buddhism: 0, christianity: 0, islam: 0 };

  // 質問を表示する関数
  function showQuestion() {
    const q = questions[currentQuestionIndex];
    document.getElementById("question-number").textContent = `Q${currentQuestionIndex + 1} / ${questions.length}`;
    document.getElementById("question").textContent = q.text;
    
    const choicesDiv = document.getElementById("choices");
    choicesDiv.innerHTML = ""; // 前の選択肢を消す
    
    q.choices.forEach(choice => {
      const btn = document.createElement("button");
      btn.textContent = choice.text;
      btn.onclick = () => selectAnswer(choice.scores);
      choicesDiv.appendChild(btn);
    });
  }

  // 選択肢を選んだ時の処理
  function selectAnswer(scores) {
    totalScores.buddhism += scores.buddhism;
    totalScores.christianity += scores.christianity;
    totalScores.islam += scores.islam;
    
    currentQuestionIndex++;
    if (currentQuestionIndex < questions.length) {
      showQuestion(); // 次の質問へ
    } else {
      showResult(); // 終わったら結果発表へ
    }
  }

  // 結果を発表する関数
  function showResult() {
    document.getElementById("quiz").style.display = "none";
    document.getElementById("result").style.display = "block";
    
    // 一番点数が高いものを探す
    let maxScore = -1;
    let finalResult = "";
    for (const [rel, score] of Object.entries(totalScores)) {
      if (score > maxScore) {
        maxScore = score;
        finalResult = rel;
      }
    }
    
    document.getElementById("result-name").textContent = results[finalResult].name;
    document.getElementById("result-desc").textContent = results[finalResult].desc;
  }

  // 最初の質問を表示してスタート
  showQuestion();
</script>
